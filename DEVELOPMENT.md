```bash
# 安装依赖
nvm use
npm install

# 启动开发模式 (CRXJS + Vite，加载 dist 作为解压扩展)
npm run dev

# 可选：保留旧式 watch 构建
npm run dev:watch

# 在另一个终端运行类型检查
npm run type-check

# 运行 ESLint
npm run lint

# 自动修复可修复的问题
npm run lint:fix

# 统一代码格式
npm run format

# 检查格式是否符合规范
npm run format:check

# 生产构建
npm run build

# 清理构建文件
npm run clean
```
